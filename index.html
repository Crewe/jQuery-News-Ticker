How to use it

<!-- in <body> -->
<ul id="js-news" class="js-hidden">
    <li class="news-item">
        <img src="img/stm32f103.png" alt=""/>
        <a href="#">Image 1</a>
    </li>
    <li class="news-item">
        <img src="img/luieee-main.png" alt=""/>
        <a href="#">Image 2</a>
    </li>
</ul>

<!-- before </body> -->
<script type="text/javascript">
    $(function () {
        $('#js-news').ticker({
            titleText: '',
            displayType: 'fade'
         });
    });
</script>
